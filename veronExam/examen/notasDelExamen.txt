---------------GENERANDO NUEVA TABLA  'alumnos';


CREATE TABLE alumnos(
id_alumno tinyint(8) unsigned AUTO_INCREMENT NOT NULL PRIMARY KEY,
nombre varchar(20)NOT NULL,
apellido varchar(20)NOT NULL,
amonestacion INT(30)NOT NULL);

------------------CREANDO TABLA materia;

id_materia tinyint(8) unsigned PRIMARY KEY NOT NULL ,
nombre_asignatura varchar(20)NOT NULL,
calificacion tinyint(8) NOT NULL,
